// metodo construtor chamado automaticamente
Aviao::Aviao(int tp){
  if (tp == 1) {
    velMax = 800;
    tipo = "Jato";
  
  }else if(tp == 2){
    velMax = 350;
    tipo = "Monomotor";
  
  } else if (tp == 3){
    velMax = 180;
    tipo = "Planador";
  }
}

void Aviao::Imprimir(){
  std::cout << "Tipo:..... " << tipo << std::endl;
  std::cout << "Vel MÃ¡xima:..... " << velMax << std::endl;
  std::cout << "Vel Atual:..... " << vel << std::endl;

}
